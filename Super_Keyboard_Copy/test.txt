#include <stdio.h> 

#include <math.h> 


int 
fun(int s, int t)
{
    int total = 0;
    if(t==0)
    {
        total = s*60;
    }
    else if(s==0)
    {
        total = t*45;
    }
    else if(s==t)
    {
        total = (s>=50) ? s*80 : s*90;
    }
    else if(s>t)
    {
        int d = s-t;
        if((s-d)>=50)
        {
            total = d*60+(s-d)*80;
        }
        else
        {
            total = d*60+(s-d)*90;
        }
    }
    else if(s<t)
    {
        int d = t-s;
        if((t-d)>=50)
        {
            total = d*45+(t-d)*80;
        }
        else
        {
            total = d*45+(t-d)*90;
        }
    }

    return total;
}

int 
main()
{
    int s, t;
    scanf("%d %d", &s, &t);

    printf("收款总额为%d", fun(s, t));

    return 0;
}
